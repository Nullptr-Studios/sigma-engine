15#pragma region // 2D Line Segment Structure
16// Holds data for representing a 2D Line Segment.
17// Data Members:
18// - mP0 mP1 : Endpoints of the line segment
19// - mN : Outward normal to the edge vector e = (P1-P0)
20// - mNdotP0 : internal use only. Value of the dot product : mN . mP0
21//
22struct AELineSegment2D
23{
24 AEVec2 mP0; // Point on the line
25 AEVec2 mP1; // Point on the line
26 AEVec2 mN; // Line's normal
27 float mNdotP0; // To avoid computing it every time it's needed
28};
