25enum AEGfxBlendMode
26{
27 AE_GFX_BM_NONE = 0, // Replaces the frame buffer color with the triangle pixel
28 AE_GFX_BM_BLEND, // Blends the fb color with the tri color using the alpha of the tri: FinalColor = TriColor * TriAlpha + FBColor * (1-TriAlpha)
29 AE_GFX_BM_ADD, // Adds the two colors together : FinalColor = TriColor + FBColor
30 AE_GFX_BM_ADD_BLENDALPHA, // Adds the two colors together and Multiplies the triangle color with its alpha: FinalColor = TriColor * TriAlpha + FBColor
31 AE_GFX_BM_MULTIPLY, // Multiplies the two colors together: FinalColor = TriColor * FBColor
32 AE_GFX_BM_NUM // keeps count of how many blend modes are implemented - also the value returned by AEGfxGetBlendMode if the arguments are set manually
33};
34
35// use those in order to have better control over the blending operation
36enum AEGfxBlendArgument
37{
38 AE_GFX_BA_ONE = 0, // 1
39 AE_GFX_BA_ZERO, // 0
40 AE_GFX_BA_SRCCOLOR, // RGB Color of the triangle pixel
41 AE_GFX_BA_SRCALPHA, // Alpha of the triangle pixel
42 AE_GFX_BA_INVSRCALPHA, // 1-Alpha of the triangle pixel
43 AE_GFX_BA_DESTCOLOR, // RGB color of the pixel currently in the frame buffer
44 AE_GFX_BA_DESTALPHA, // Alpha of the pixel currently in the frame buffer
45 AE_GFX_BA_INVDESTALPHA, // 1-Alpha of the pixel currently in the frame buffer
46 AE_GFX_BA_NUM
47};
48
49// Texture filters define the behavior of sampling a texture when the size on screen
50// of the object to draw is greater or smaller than the texture size.
51enum AEGfxTextureFilter
52{
53 AE_GFX_TF_NEAREST = 0, // uses the nearest pixel to the specified texture coordinates
54 AE_GFX_TF_LINEAR, // uses the weighted average of the texture elements that are closest to the specified texture coordinates.
55 AE_GFX_TF_NUM
56};
